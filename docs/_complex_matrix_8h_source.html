55 bool getLUDecomposition(ComplexMatrix<T>& Decomposition, ComplexMatrix<T>& Permutations) {
56 // Верхня, нижня трикутні матриці та матриця перестановок
57 if (this->columns != this->rows) {
58 printError(CANNOT_FIND_LU_DECOMPOSITION_ERROR_CODE);
59 return false;
60 }
61
62 // Верхня трикутна матриця, нижня трикутні матриці
63 // зберігаються в Decomposition
64 // Decomposition = Lower + Upper - E
65 // E - одинична матриця
66 Decomposition = ComplexMatrix(*this),
67 Permutations = ComplexMatrix::getIdentity(this->rows);
68
69 for (int i = 0; i < rows; i++) {
70 // Пошук опорного елемента
71 int supportElement = i;
72 while (supportElement < this->rows && Decomposition[supportElement][i] == 0) {
73 supportElement++;
74 }
75 if (supportElement == this->rows) {
76 printError(CANNOT_FIND_LU_DECOMPOSITION_ERROR_CODE);
77 return false;
78 }
79
80 // Міняємо місцями i-ту рядок і рядок з опорним елементом
81 Permutations.swapRows(supportElement, i);
82 Decomposition.swapRows(supportElement, i);
83 for (int j = i + 1; j < rows; j++) {
84 Decomposition[j][i] /= Decomposition[i][i];
85 for (int k = i + 1; k < rows; k++)
86 Decomposition[j][k] -= Decomposition[j][i] * Decomposition[i][k];
87 }
88 }
89 return true;
90 }

98 void StrassenMultiplication(ComplexMatrix<T> A, ComplexMatrix<T> B, ComplexMatrix<T>& Product) {
99 // База
100 if (A.rows <= 64 || A.columns || B.columns) {
101 Product = A * B;
102 return;
103 }
104 int newSize = A.rows / 2;

108 c11(newSize), c12(newSize), c21(newSize), c22(newSize),

110 p5(newSize), p6(newSize), p7(newSize);
111
112 // Ділимо матриці на підматриці
113 for (int i = 0; i < newSize; i++) {
114 for (int j = 0; j < newSize; j++) {
115 a11[i][j] = A[i][j];
116 a12[i][j] = A[i][j + newSize];
117 a21[i][j] = A[i + newSize][j];
118 a22[i][j] = A[i + newSize][j + newSize];
119
120 b11[i][j] = B[i][j];
121 b12[i][j] = B[i][j + newSize];
122 b21[i][j] = B[i + newSize][j];
123 b22[i][j] = B[i + newSize][j + newSize];
124 }
125 }
126
127 StrassenMultiplication(a11, b12 - b22, p1);
128 StrassenMultiplication(a11 + a12, b22, p2);
129 StrassenMultiplication(a21 + a22, b11, p3);
130 StrassenMultiplication(a22, b21 - b11, p4);
131 StrassenMultiplication(a11 + a22, b11 + b22, p5);
132 StrassenMultiplication(a12 - a22, b21 + b22, p6);
133 StrassenMultiplication(a11 - a21, b11 + b12, p7);
134
135 c11 = p4 + p5 + p6 - p2;
136 c12 = p1 + p2;
137 c21 = p3 + p4;
138 c22 = p1 + p5 - p3 - p7;
139
140 // Складаємо результат у єдину матрицю:
141 for (int i = 0; i < newSize; i++) {

143 Product[i][j] = c11[i][j];
144 Product[i][j + newSize] = c12[i][j];

156 ComplexMatrix<T> getInverseGaussJordan() {
157 // Якщо матриця не квадратна
158 if (this->columns != this->rows) {
159 printError(MATRIX_IS_NOT_SQUARE_ERROR_CODE);
160 return *this;
161 }
162
163 // Одинична матриця
164 ComplexMatrix<T> Identity = ComplexMatrix<T>::getIdentity(this->rows);
165 // Копія, щоб не змінювати реальну
166 ComplexMatrix<T> MatrixCopy = ComplexMatrix(*this);
167
168 for (int i = 0; i < this->rows; i++) {
169 // Якщо 0, то шукаємо перший ненульовий і свапаємо
170 if (MatrixCopy[i][i] == 0) {
171 int supportElement = i + 1;
172 while (supportElement < this->rows && MatrixCopy[supportElement][i] == 0) {
173 supportElement++;
174 }
175 if (supportElement == this->rows) {
176 printError(CANNOT_FIND_INVERSE_MATRIX_ERROR_CODE);
177 return *this;
178 }
179 MatrixCopy.swapRows(i, supportElement);
180 Identity.swapRows(i, supportElement);
181 }
182
183 // Ділимо на a + bi
184 Complex<T> divider = MatrixCopy[i][i];
185
186 for (int j = 0; j < this->rows; j++) {
187 Identity[i][j] /= divider;
188 }
189 for (int j = i + 1; j < this->rows; j++) {
190 MatrixCopy[i][j] /= divider;
191 }
192
193 // Елементи під діагоналлю в 0
194 for (int k = i + 1; k < this->rows; k++) {
195 Complex<T> factor = MatrixCopy[k][i];
196
197 for (int j = 0; j < this->rows; j++) {
198 if (j >= i) MatrixCopy[k][j] -= (MatrixCopy[i][j] * factor);
199 Identity[k][j] -= (Identity[i][j] * factor);
200 }
201 }
202
203 // Елементи над діагоналлю в 0
204 for (int k = i - 1; k >= 0; k--) {
205 Complex<T> factor = MatrixCopy[k][i];
206
207 for (int j = 0; j < this->rows; j++) {
208 if (j >= i) MatrixCopy[k][j] -= (MatrixCopy[i][j] * factor);
209 Identity[k][j] -= (Identity[i][j] * factor);
210 }
211 }
212 }
213 return Identity;
214 }

221 ComplexMatrix<T> getInverseLU() {
222 if (this->columns != this->rows) {
223 printError(MATRIX_IS_NOT_SQUARE_ERROR_CODE);
224 return *this;
225 }
226 ComplexMatrix<T> Decomposition, Permutation;
227 if (getLUDecomposition(Decomposition, Permutation) == false) {
228 printError(CANNOT_FIND_INVERSE_MATRIX_ERROR_CODE);
229 return *this;
230 }
231
232 // Спробував через заміни, чомусь працює
233 ComplexMatrix<T> Inverse(this->rows);
234 vector<Complex<T>> D(this->rows);
235
236 for (int i = 0; i < this->rows; i++) {
237 // forward substitution
238 for (int j = 0; j < this->rows; j++) {
239 Complex<T> numerator = i == j ? Complex<T>(1, 0) : Complex<T>(0, 0);
240 for (int k = 0; k < j; k++) {
241 numerator -= Decomposition[j][k] * D[k];
242 }
243 D[j] = numerator;
244 }
245 // backward substitution
246 for (int j = this->rows - 1; j >= 0; j--) {
247 Complex<T> numerator = D[j] ;
248 for (int k = j + 1; k < this->rows; k++) {
249 numerator -= Decomposition[j][k] * Inverse[k][i];
250 }
251 Inverse[j][i] = numerator / Decomposition[j][j];
252 }
253 }
254
255 for (int i = this->rows - 1; i >= 0; i--) {
256 for (int j = 0; j < i; j++) {
257 if (Permutation[i][j] != Complex<T>(1, 0)) {
258 continue;
259 }
260
261 Permutation.swapRows(i, j);
262 Inverse.swapColumns(i, j);
263 i++;
264 break;
265 }
266 }
267
268 return Inverse;
